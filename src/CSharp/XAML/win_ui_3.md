    {
        return value;

// xml 
<Grid.Resources>
    // 必须先引用才能使用
    <local:DateFormatter x:Key="FormatConverter" />
</Grid.Resources>
<TextBlock Text="{Binding Path=ReleaseDate, Mode=OneWay,
                            // converter 源
                            Converter={StaticResource FormatConverter},
