# uniApp - 微信端

微信端的配置

# manifest 配置

"mp-weixin" : {
    /* 小程序特有相关 */
    "appid" : "小程序id",
    "setting" : {
        "urlCheck" : false,
        "postcss" : true,
        "minified" : true,
        "es6" : true
    },
    "usingComponents" : true,
    // 懒加载插件
    "lazyCodeLoading" : "requiredComponents",
        "plugins": {
        },
        "requiredBackgroundModes": [
            "audio"
        ],
        "sitemapLocation": "sitemap.json",
        "resizable": true
},

# package.json 扩张配置

{
    /**
     * package.json其它原有配置 
     * 拷贝代码后请去掉注释！
     */
    "uni-app": {// 扩展配置
        "scripts": {
            "custom-platform": { //自定义编译平台配置，可通过cli方式调用
                "title":"自定义扩展名称", // 在HBuilderX中会显示在 运行/发行 菜单中
                "browser":"",  //运行到的目标浏览器，仅当UNI_PLATFORM为h5时有效
                "env": {//环境变量
                    "UNI_PLATFORM": "",  //基准平台
                    "MY_TEST": "", // ... 其他自定义环境变量
                 },
                "define": { //自定义条件编译
                    "CUSTOM-CONST": true //自定义条件编译常量，建议为大写
                }
            }
        }    
    }
}
